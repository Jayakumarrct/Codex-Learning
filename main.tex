\section{Warm-up}
\begin{definition}
A group is a set $G$ with an operation $\cdot$ satisfying associativity, identity, and inverses.
\end{definition}

\begin{theorem}
If $G$ is a group, the identity is unique.
\end{theorem}

\begin{proof}
Standard argument: if $e,e'$ are identities, then $e=e\cdot e'=e'$.
\end{proof}
